def call(*args, &block)
  %x{
    if (block !== nil) {
      self._p = block;
    }

    var result;

    if (self.is_lambda) {
      result = self.apply(null, args);
    }
    else {
      result = Opal.$yieldX(self, args);
    }

    if (result === $breaker) {
      return $breaker.$v;
    }

    return result;
  }
end
