def each(*args, &block)
  yielder = Yielder.new(&block)

  %x{
    try {
      args.unshift(#{yielder});

      if ($opal.$yieldX(#@block, args) === $breaker) {
        return $breaker.$v;
      }
    }
    catch (e) {
      if (e === $breaker) {
        return $breaker.$v;
      }
      else {
        throw e;
      }
    }
  }

  self
end
