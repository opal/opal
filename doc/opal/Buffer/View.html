def initialize(buffer, offset = nil, length = nil)
  if native?(buffer)
    super(buffer)
  elsif offset && length
    super(`new DataView(#{buffer.to_n}, #{offset.to_n}, #{length.to_n})`)
  elsif offset
    super(`new DataView(#{buffer.to_n}, #{offset.to_n})`)
  else
    super(`new DataView(#{buffer.to_n})`)
  end

  @buffer = buffer
  @offset = offset
end
